19  /**
20   * <p> This class represents a <code>Map</code> collection with real-time 
21   *     behavior. Unless the map's size exceeds its current capacity, 
22   *     no dynamic memory allocation is ever performed and response time is 
23   *     <b>extremely fast</b> and <b>consistent</b>.</p>
24   *  
25   * <p> Our <a href="http://jade.dautelle.com/doc/benchmark.txt">benchmark</a> 
26   *     indicates that {@link FastMap#put FastMap.put(key, value)} is up to 
27   *     <b>5x faster</b> than <code>java.util.HashMap.put(key, value)</code>.
28   *     This difference is mostly due to the cost of the <code>Map.Entry</code>
29   *     allocations that {@link FastMap} avoids by recycling its entries 
30   *     (see note below).</p>
31   * 
32   * <p> {@link FastMap} has a predictable iteration order, which is the order
33   *     in which keys were inserted into the map (similar to 
34   *     <code>java.util.LinkedHashMap</code> collection class).</p>
35   * 
36   * <p> Applications may change the resizing policy  of {@link FastMap} 
37   *     by overriding the {@link #sizeChanged} method. For example, to improve
38   *     predictability, automatic resizing can be disabled.</p>  
39   *
40   * <p> This implementation is not synchronized. Multiple threads accessing
41   *     or modifying the collection must be synchronized externally.</p>
42   * 
43   * <p> <b>Note:</b> To avoid dynamic memory allocations, {@link FastMap}
44   *     maintains an internal pool of <code>Map.Entry</code> objects. The size
45   *     of the pool is determined by the map's capacity. When an entry is
46   *     removed from the map, it is automatically restored to the pool.</p>
47   * 
48   * <p><i> This class is <b>public domain</b> (not copyrighted).</i></p>
49   *  
50   * @author  <a href="mailto:jean-marie@dautelle.com">Jean-Marie Dautelle</a>
51   * @version 5.3, October 31 2003
52   */
53  public class FastMap<K,V> implements Map<K,V>, Cloneable, Serializable {

235     /**
236      * Associates the specified value with the specified key in this 
237      * {@link FastMap}. If the {@link FastMap} previously contained a mapping
238      * for this key, the old value is replaced.
239      *
240      * @param  key the key with which the specified value is to be associated.
241      * @param  value the value to be associated with the specified key.
242      * @return the previous value associated with specified key,
243      *         or <code>null</code> if there was no mapping for key.
244      *         A <code>null</code> return can also indicate that the map 
245      *         previously associated <code>null</code> with the specified key.
246      * @throws NullPointerException if the key is <code>null</code>.
247      */
248     public Object put(Object key, Object value) {
249         EntryImpl entry = _entries[keyHash(key) & _mask];
250         while (entry != null) {
251             if (key.equals(entry._key) ) {
252                 Object prevValue = entry._value; 
253                 entry._value = value;
254                 return prevValue;
255             }
256             entry = entry._next;
257         }
258         // No previous mapping.
259         addEntry(key, value);
260         return null;
261     }

264      * Copies all of the mappings from the specified map to this 
265      * {@link FastMap}.
266      *
267      * @param  map the mappings to be stored in this map.
268      * @throws NullPointerException the specified map is <code>null</code>, or
269      *         the specified map contains <code>null</code> keys.

271     public void putAll(Map<? extends K, ? extends V> map) {
272         for ( Entry<? extends K, ? extends V> entry : map.entrySet() )
273         {
274             addEntry( entry.getKey(), entry.getValue() );
275         }

278     /**
279      * Removes the mapping for this key from this {@link FastMap} if present.
280      *
281      * @param  key the key whose mapping is to be removed from the map.
282      * @return previous value associated with specified key,
283      *         or <code>null</code> if there was no mapping for key.
284      *         A <code>null</code> return can also indicate that the map 
285      *         previously associated <code>null</code> with the specified key.
286      * @throws NullPointerException if the key is <code>null</code>.
287      */
288     public V remove(Object key) {
289         EntryImpl<K,V> entry = _entries[keyHash(key) & _mask];
290         while (entry != null) {
291             if (key.equals(entry._key) ) {
292                 V prevValue = entry._value;
293                 removeEntry(entry);
294                 return prevValue;
295             }
296             entry = entry._next;
297         }
298         return null;
299     }

301     /**
302      * Removes all mappings from this {@link FastMap}.
303      */
304     public void clear() {
305         // Clears all keys, values and buckets linked lists.
306         for (EntryImpl entry = _mapFirst; entry != null; entry = entry._after) {
307             entry._key = null;
308             entry._value = null;
309             entry._before = null;
310             entry._next = null;
311             if (entry._previous == null) { // First in bucket.
312                 _entries[entry._index] = null;
313             } else {
314                 entry._previous = null;
315             }
316         }
317         
318         // Recycles all entries.
319         if (_mapLast != null) {
320             _mapLast._after = _poolFirst; // Connects to pool.
321             _poolFirst = _mapFirst;
322             _mapFirst = null;
323             _mapLast = null;
324             _size = 0;
325             sizeChanged();
326         }
327     }

329     /**
330      * Changes the current capacity of this {@link FastMap}. If the capacity
331      * is increased, new entries are allocated and added to the pool. 
332      * If the capacity is decreased, entries from the pool are deallocated 
333      * (and are eventually garbage collected). The capacity also determined 
334      * the number of buckets for the hash table.
335      * 
336      * @param newCapacity the new capacity of this map.
337      */
338     public void setCapacity(int newCapacity) {
339         if (newCapacity > _capacity) { // Capacity increases.
340             for (int i = _capacity; i < newCapacity; i++) {
341                 EntryImpl entry = new EntryImpl();
342                 entry._after = _poolFirst;
343                 _poolFirst = entry;
344             }
345         } else if (newCapacity < _capacity) { // Capacity decreases.
346             for (    int i = newCapacity; 
347                      (i < _capacity) && (_poolFirst != null); i++) {
348                 // Disconnects the entry for gc to do its work.
349                 EntryImpl entry = _poolFirst;
350                 _poolFirst = entry._after;
351                 entry._after = null; // All pointers are now null!
352             }
353         }
354         // Find a power of 2 >= capacity
355         int tableLength = 16;
356         while (tableLength < newCapacity) {
357             tableLength <<= 1;
358         }
359         // Checks if the hash table has to be re-sized.
360         if (_entries.length != tableLength) {
361             _entries = new EntryImpl[tableLength];
362             _mask = tableLength - 1;
363             
364             // Repopulates the hash table.
365             EntryImpl entry = _mapFirst;
366             while (entry != null) {
367                 int index = keyHash(entry._key) & _mask;
368                 entry._index = index;
369         
370                 // Connects to bucket.
371                 entry._previous = null; // Resets previous.
372                 EntryImpl next = _entries[index];
373                 entry._next = next;
374                 if (next != null) {
375                     next._previous = entry;
376                 }
377                 _entries[index] = entry;
378                 
379                 entry = entry._after;
380             }
381         }
382         _capacity = newCapacity;
383     }

403     /**
404      * Compares the specified object with this {@link FastMap} for equality. 
405      * Returns <code>true</code> if the given object is also a map and the two
406      * maps represent the same mappings (regardless of collection iteration
407      * order).
408      *
409      * @param obj the object to be compared for equality with this map.
410      * @return <code>true</code> if the specified object is equal to this map;
411      *         <code>false</code> otherwise.
412      */
413     public boolean equals(Object obj) {
414         if (obj == this) {
415             return true;
416         } else if (obj instanceof Map) {
417             Map that = (Map) obj;
418             if (this.size() == that.size()) {
419                 EntryImpl entry = _mapFirst;
420                 while (entry != null) {
421                     if (!that.entrySet().contains(entry)) {
422                         return false;
423                     }
424                     entry = entry._after;
425                 }
426                 return true;
427             } else {
428                 return false;
429             }
430         } else {
431             return false;
432         }
433     }

615     /**
616      * This methods is being called when the size of this {@link FastMap} 
617      * has changed. The default behavior is to double the map's capacity 
618      * when the map's size reaches the current map's capacity.
619      * Sub-class may override this method to implement custom resizing 
620      * policies or to disable automatic resizing. For example:<pre>
621      *     Map fixedCapacityMap = new FastMap(256) { 
622      *           protected sizeChanged() {
623      *               // Do nothing, automatic resizing disabled.
624      *           }
625      *     };</pre>
626      * @see #setCapacity
627      */
628     protected void sizeChanged() {
629         if (size() > capacity()) {
630             setCapacity(capacity() * 2);
631         }

634     /**
635      * Returns the hash code for the specified key. The formula being used 
636      * is identical to the formula used by <code>java.util.HashMap</code>
637      * (ensures similar behavior for ill-conditioned hashcode keys).
638      * 
639      * @param key the key to calculate the hashcode for.
640      * @return the hash code for the specified key.
641      */
642     private static int keyHash(Object key) {
643         // From HashMap.hash(Object) function.
644         int hashCode = key.hashCode();
645         hashCode += ~(hashCode << 9);
646         hashCode ^=  (hashCode >>> 14);
647         hashCode +=  (hashCode << 4);
648         hashCode ^=  (hashCode >>> 10);
649         return hashCode;
650     }

652     /**
653      * Adds a new entry for the specified key and value.
654      * @param key the entry's key.
655      * @param value the entry's value.
656      */
657     private void addEntry(Object key, Object value) {
658         EntryImpl entry = _poolFirst;
659         if (entry != null) {
660             _poolFirst = entry._after;
661             entry._after = null;
662         } else { // Pool empty.
663             entry = new EntryImpl();
664         }
665         
666         // Setup entry paramters.
667         entry._key = key;
668         entry._value = value;
669         int index = keyHash(key) & _mask;
670         entry._index = index;
671         
672         // Connects to bucket.
673         EntryImpl next = _entries[index];
674         entry._next = next;
675         if (next != null) {
676             next._previous = entry;
677         }
678         _entries[index] = entry;
679         
680         // Connects to collection.
681         if (_mapLast != null) {
682             entry._before = _mapLast;
683             _mapLast._after = entry;
684         } else {
685             _mapFirst = entry;
686         }
687         _mapLast = entry;
688         
689         // Updates size.
690         _size++;
691         sizeChanged();
692     }

694     /**
695      * Removes the specified entry from the map.
696      * 
697      * @param entry the entry to be removed.
698      */
699     private void removeEntry(EntryImpl entry) {
700         
701         // Removes from bucket.
702         EntryImpl previous = entry._previous;
703         EntryImpl next = entry._next;
704         if (previous != null) {
705             previous._next = next;
706             entry._previous = null;
707         } else { // First in bucket.
708             _entries[entry._index] = next;
709         }
710         if (next != null) { 
711             next._previous = previous;
712             entry._next = null;
713         } // Else do nothing, no last pointer.
714         
715         // Removes from collection.
716         EntryImpl before = entry._before;
717         EntryImpl after = entry._after;
718         if (before != null) { 
719             before._after = after;
720             entry._before = null;
721         } else { // First in collection.
722             _mapFirst = after;
723         }
724         if (after != null) { 
725             after._before = before;
726         } else { // Last in collection.
727             _mapLast = before;
728         }
729 
730         // Clears value and key.
731         entry._key = null;
732         entry._value = null;
733 
734         // Recycles.
735         entry._after = _poolFirst;
736         _poolFirst = entry;
737         
738         // Updates size.
739         _size--;
740         sizeChanged();
741     }

744      * Initializes this instance for the specified capacity.
