24   * Format of Internet Message Bodies</cite> by Freed and Borenstein.</p>

37      /**
38       * Chunk size per RFC 2045 section 6.8.
39       *
40       * <p>The {@value} character limit does not count the trailing CRLF, but counts
41       * all other characters, including any equal signs.</p>
42       *
43       * @see <a href="http://www.ietf.org/rfc/rfc2045.txt">RFC 2045 section 6.8</a>
44       */
45      static final int CHUNK_SIZE = 76;
46  
47      /**
48       * Chunk separator per RFC 2045 section 2.1.
49       *
50       * @see <a href="http://www.ietf.org/rfc/rfc2045.txt">RFC 2045 section 2.1</a>
51       */
52      static final byte[] CHUNK_SEPARATOR = "\r\n".getBytes();
53  
54      /**
55       * The base length.
56       */
57      static final int BASELENGTH = 255;
58  
59      /**
60       * Lookup length.
61       */
62      static final int LOOKUPLENGTH = 64;
63  
64      /**
65       * Used to calculate the number of bits in a byte.
66       */
67      static final int EIGHTBIT = 8;
68  
69      /**
70       * Used when encoding something which has fewer than 24 bits.
71       */
72      static final int SIXTEENBIT = 16;
73  
74      /**
75       * Used to determine how many bits data contains.
76       */
77      static final int TWENTYFOURBITGROUP = 24;
78  
79      /**
80       * Used to get the number of Quadruples.
81       */
82      static final int FOURBYTE = 4;
83  
84      /**
85       * Used to test the sign of a byte.
86       */
87      static final int SIGN = -128;

89      /**
90       * Byte used to pad output.
91       */
92      static final byte PAD = (byte) '=';

94      /**
95       * Contains the Base64 values <code>0</code> through <code>63</code> accessed by using character encodings as
96       * indices.
97       * <p>
98       * For example, <code>base64Alphabet['+']</code> returns <code>62</code>.
99       * </p>
100      * <p>
101      * The value of undefined encodings is <code>-1</code>.
102      * </p>
103      */
104     private static byte[] base64Alphabet = new byte[BASELENGTH];
105 
106     /**
107      * <p>
108      * Contains the Base64 encodings <code>A</code> through <code>Z</code>, followed by <code>a</code> through
109      * <code>z</code>, followed by <code>0</code> through <code>9</code>, followed by <code>+</code>, and
110      * <code>/</code>.
111      * </p>
112      * <p>
113      * This array is accessed by using character values as indices.
114      * </p>
115      * <p>
116      * For example, <code>lookUpBase64Alphabet[62] </code> returns <code>'+'</code>.
117      * </p>
118      */
119     private static byte[] lookUpBase64Alphabet = new byte[LOOKUPLENGTH];
120 
121     // Populating the lookup and character arrays
122     static {
123         for (int i = 0; i < BASELENGTH; i++) {
124             base64Alphabet[i] = (byte) -1;
125         }
126         for (int i = 'Z'; i >= 'A'; i--) {
127             base64Alphabet[i] = (byte) (i - 'A');
128         }
129         for (int i = 'z'; i >= 'a'; i--) {
130             base64Alphabet[i] = (byte) (i - 'a' + 26);
131         }
132         for (int i = '9'; i >= '0'; i--) {
133             base64Alphabet[i] = (byte) (i - '0' + 52);
134         }
135 
136         base64Alphabet['+'] = 62;
137         base64Alphabet['/'] = 63;
138 
139         for (int i = 0; i <= 25; i++) {
140             lookUpBase64Alphabet[i] = (byte) ('A' + i);
141         }
142 
143         for (int i = 26, j = 0; i <= 51; i++, j++) {
144             lookUpBase64Alphabet[i] = (byte) ('a' + j);
145         }
146 
147         for (int i = 52, j = 0; i <= 61; i++, j++) {
148             lookUpBase64Alphabet[i] = (byte) ('0' + j);
149         }
150 
151         lookUpBase64Alphabet[62] = (byte) '+';
152         lookUpBase64Alphabet[63] = (byte) '/';
153     }

155     /**
156      * Returns whether or not the <code>octect</code> is in the base 64 alphabet.
157      *
158      * @param octect The value to test
159      * @return <code>true</code> if the value is defined in the the base 64 alphabet, <code>false</code> otherwise.
160      */
161     private static boolean isBase64(byte octect) {
162         if (octect == PAD) {
163             return true;
164         } else if (octect < 0 || base64Alphabet[octect] == -1) {
165             return false;
166         } else {
167             return true;
168         }
169     }

171     /**
172      * Tests a given byte array to see if it contains
173      * only valid characters within the Base64 alphabet.
174      *
175      * @param arrayOctect byte array to test
176      * @return <code>true</code> if all bytes are valid characters in the Base64
177      *         alphabet or if the byte array is empty; false, otherwise
178      */
179     public static boolean isArrayByteBase64(byte[] arrayOctect) {
180 
181         arrayOctect = discardWhitespace(arrayOctect);
182 
183         int length = arrayOctect.length;
184         if (length == 0) {
185             // shouldn't a 0 length array be valid base64 data?
186             // return false;
187             return true;
188         }
189         for ( byte anArrayOctect : arrayOctect )
190         {
191             if ( !isBase64( anArrayOctect ) )
192             {
193                 return false;
194             }
195         }
196         return true;
197     }

233     /**
234      * Encodes binary data using the base64 algorithm, optionally
235      * chunking the output into 76 character blocks.
236      *
237      * @param binaryData Array containing binary data to encode.
238      * @param isChunked if <code>true</code> this encoder will chunk
239      *                  the base64 output into 76 character blocks
240      * @return Base64-encoded data.
241      */
242     public static byte[] encodeBase64(byte[] binaryData, boolean isChunked) {
243         int lengthDataBits = binaryData.length * EIGHTBIT;
244         int fewerThan24bits = lengthDataBits % TWENTYFOURBITGROUP;
245         int numberTriplets = lengthDataBits / TWENTYFOURBITGROUP;
246         byte encodedData[] = null;
247         int encodedDataLength = 0;
248         int nbrChunks = 0;
249 
250         if (fewerThan24bits != 0) {
251             //data not divisible by 24 bit
252             encodedDataLength = (numberTriplets + 1) * 4;
253         } else {
254             // 16 or 8 bit
255             encodedDataLength = numberTriplets * 4;
256         }
257 
258         // If the output is to be "chunked" into 76 character sections,
259         // for compliance with RFC 2045 MIME, then it is important to
260         // allow for extra length to account for the separator(s)
261         if (isChunked) {
262 
263             nbrChunks =
264                 (CHUNK_SEPARATOR.length == 0 ? 0 : (int) Math.ceil((float) encodedDataLength / CHUNK_SIZE));
265             encodedDataLength += nbrChunks * CHUNK_SEPARATOR.length;
266         }
267 
268         encodedData = new byte[encodedDataLength];
269 
270         byte k = 0, l = 0, b1 = 0, b2 = 0, b3 = 0;
271 
272         int encodedIndex = 0;
273         int dataIndex = 0;
274         int i = 0;
275         int nextSeparatorIndex = CHUNK_SIZE;
276         int chunksSoFar = 0;
277 
278         //log.debug("number of triplets = " + numberTriplets);
279         for (i = 0; i < numberTriplets; i++) {
280             dataIndex = i * 3;
281             b1 = binaryData[dataIndex];
282             b2 = binaryData[dataIndex + 1];
283             b3 = binaryData[dataIndex + 2];
284 
285             //log.debug("b1= " + b1 +", b2= " + b2 + ", b3= " + b3);
286 
287             l = (byte) (b2 & 0x0f);
288             k = (byte) (b1 & 0x03);
289 
290             byte val1 =
291                 ((b1 & SIGN) == 0) ? (byte) (b1 >> 2) : (byte) ((b1) >> 2 ^ 0xc0);
292             byte val2 =
293                 ((b2 & SIGN) == 0) ? (byte) (b2 >> 4) : (byte) ((b2) >> 4 ^ 0xf0);
294             byte val3 =
295                 ((b3 & SIGN) == 0) ? (byte) (b3 >> 6) : (byte) ((b3) >> 6 ^ 0xfc);
296 
297             encodedData[encodedIndex] = lookUpBase64Alphabet[val1];
298             //log.debug( "val2 = " + val2 );
299             //log.debug( "k4   = " + (k<<4) );
300             //log.debug(  "vak  = " + (val2 | (k<<4)) );
301             encodedData[encodedIndex + 1] =
302                 lookUpBase64Alphabet[val2 | (k << 4)];
303             encodedData[encodedIndex + 2] =
304                 lookUpBase64Alphabet[(l << 2) | val3];
305             encodedData[encodedIndex + 3] = lookUpBase64Alphabet[b3 & 0x3f];
306 
307             encodedIndex += 4;
308 
309             // If we are chunking, let's put a chunk separator down.
310             if (isChunked) {
311                 // this assumes that CHUNK_SIZE % 4 == 0
312                 if (encodedIndex == nextSeparatorIndex) {
313                     System.arraycopy(
314                         CHUNK_SEPARATOR,
315                         0,
316                         encodedData,
317                         encodedIndex,
318                         CHUNK_SEPARATOR.length);
319                     chunksSoFar++;
320                     nextSeparatorIndex =
321                         (CHUNK_SIZE * (chunksSoFar + 1)) +
322                         (chunksSoFar * CHUNK_SEPARATOR.length);
323                     encodedIndex += CHUNK_SEPARATOR.length;
324                 }
325             }
326         }
327 
328         // form integral number of 6-bit groups
329         dataIndex = i * 3;
330 
331         if (fewerThan24bits == EIGHTBIT) {
332             b1 = binaryData[dataIndex];
333             k = (byte) (b1 & 0x03);
334             //log.debug("b1=" + b1);
335             //log.debug("b1<<2 = " + (b1>>2) );
336             byte val1 =
337                 ((b1 & SIGN) == 0) ? (byte) (b1 >> 2) : (byte) ((b1) >> 2 ^ 0xc0);
338             encodedData[encodedIndex] = lookUpBase64Alphabet[val1];
339             encodedData[encodedIndex + 1] = lookUpBase64Alphabet[k << 4];
340             encodedData[encodedIndex + 2] = PAD;
341             encodedData[encodedIndex + 3] = PAD;
342         } else if (fewerThan24bits == SIXTEENBIT) {
343 
344             b1 = binaryData[dataIndex];
345             b2 = binaryData[dataIndex + 1];
346             l = (byte) (b2 & 0x0f);
347             k = (byte) (b1 & 0x03);
348 
349             byte val1 =
350                 ((b1 & SIGN) == 0) ? (byte) (b1 >> 2) : (byte) ((b1) >> 2 ^ 0xc0);
351             byte val2 =
352                 ((b2 & SIGN) == 0) ? (byte) (b2 >> 4) : (byte) ((b2) >> 4 ^ 0xf0);
353 
354             encodedData[encodedIndex] = lookUpBase64Alphabet[val1];
355             encodedData[encodedIndex + 1] =
356                 lookUpBase64Alphabet[val2 | (k << 4)];
357             encodedData[encodedIndex + 2] = lookUpBase64Alphabet[l << 2];
358             encodedData[encodedIndex + 3] = PAD;
359         }
360 
361         if (isChunked) {
362             // we also add a separator to the end of the final chunk.
363             if (chunksSoFar < nbrChunks) {
364                 System.arraycopy(
365                     CHUNK_SEPARATOR,
366                     0,
367                     encodedData,
368                     encodedDataLength - CHUNK_SEPARATOR.length,
369                     CHUNK_SEPARATOR.length);
370             }
371         }
372 
373         return encodedData;
374     }

376     /**
377      * Decodes Base64 data into octects
378      *
379      * @param base64Data Byte array containing Base64 data
380      * @return Array containing decoded data.
381      */
382     public static byte[] decodeBase64(byte[] base64Data) {
383         // RFC 2045 requires that we discard ALL non-Base64 characters
384         base64Data = discardNonBase64(base64Data);
385 
386         // handle the edge case, so we don't have to worry about it later
387         if (base64Data.length == 0) {
388             return new byte[0];
389         }
390 
391         int numberQuadruple = base64Data.length / FOURBYTE;
392         byte decodedData[] = null;
393         byte b1 = 0, b2 = 0, b3 = 0, b4 = 0, marker0 = 0, marker1 = 0;
394 
395         // Throw away anything not in base64Data
396 
397         int encodedIndex = 0;
398         int dataIndex = 0;
399         {
400             // this sizes the output array properly - rlw
401             int lastData = base64Data.length;
402             // ignore the '=' padding
403             while (base64Data[lastData - 1] == PAD) {
404                 if (--lastData == 0) {
405                     return new byte[0];
406                 }
407             }
408             decodedData = new byte[lastData - numberQuadruple];
409         }
410 
411         for (int i = 0; i < numberQuadruple; i++) {
412             dataIndex = i * 4;
413             marker0 = base64Data[dataIndex + 2];
414             marker1 = base64Data[dataIndex + 3];
415 
416             b1 = base64Alphabet[base64Data[dataIndex]];
417             b2 = base64Alphabet[base64Data[dataIndex + 1]];
418 
419             if (marker0 != PAD && marker1 != PAD) {
420                 //No PAD e.g 3cQl
421                 b3 = base64Alphabet[marker0];
422                 b4 = base64Alphabet[marker1];
423 
424                 decodedData[encodedIndex] = (byte) (b1 << 2 | b2 >> 4);
425                 decodedData[encodedIndex + 1] =
426                     (byte) (((b2 & 0xf) << 4) | ((b3 >> 2) & 0xf));
427                 decodedData[encodedIndex + 2] = (byte) (b3 << 6 | b4);
428             } else if (marker0 == PAD) {
429                 //Two PAD e.g. 3c[Pad][Pad]
430                 decodedData[encodedIndex] = (byte) (b1 << 2 | b2 >> 4);
431             } else if (marker1 == PAD) {
432                 //One PAD e.g. 3cQ[Pad]
433                 b3 = base64Alphabet[marker0];
434 
435                 decodedData[encodedIndex] = (byte) (b1 << 2 | b2 >> 4);
436                 decodedData[encodedIndex + 1] =
437                     (byte) (((b2 & 0xf) << 4) | ((b3 >> 2) & 0xf));
438             }
439             encodedIndex += 3;
440         }
441         return decodedData;
442     }

444     /**
445      * Discards any whitespace from a base-64 encoded block.
446      *
447      * @param data The base-64 encoded data to discard the whitespace
448      * from.
449      * @return The data, less whitespace (see RFC 2045).
450      */
451     static byte[] discardWhitespace(byte[] data) {
452         byte groomedData[] = new byte[data.length];
453         int bytesCopied = 0;
454 
455         for ( byte aData : data )
456         {
457             switch ( aData )
458             {
459                 case (byte) ' ':
460                 case (byte) '\n':
461                 case (byte) '\r':
462                 case (byte) '\t':
463                     break;
464                 default:
465                     groomedData[bytesCopied++] = aData;
466             }
467         }
468 
469         byte packedData[] = new byte[bytesCopied];
470 
471         System.arraycopy(groomedData, 0, packedData, 0, bytesCopied);
472 
473         return packedData;
474     }

476     /**
477      * Discards any characters outside of the base64 alphabet, per
478      * the requirements on page 25 of RFC 2045 - "Any characters
479      * outside of the base64 alphabet are to be ignored in base64
480      * encoded data."
481      *
482      * @param data The base-64 encoded data to groom
483      * @return The data, less non-base64 characters (see RFC 2045).
484      */
485     static byte[] discardNonBase64(byte[] data) {
486         byte groomedData[] = new byte[data.length];
487         int bytesCopied = 0;
488 
489         for ( byte aData : data )
490         {
491             if ( isBase64( aData ) )
492             {
493                 groomedData[bytesCopied++] = aData;
494             }
495         }
496 
497         byte packedData[] = new byte[bytesCopied];
498 
499         System.arraycopy(groomedData, 0, packedData, 0, bytesCopied);
500 
501         return packedData;
502     }
