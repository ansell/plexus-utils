13  /**
14   * <p> This class provides cache access to <code>Map</code> collections.</p>
15   * 
16   * <p> Instance of this class can be used as "proxy" for any collection
17   *     implementing the <code>java.util.Map</code> interface.</p>
18   * 
19   * <p> Typically, {@link CachedMap} are used to accelerate access to
20   *     large collections when the access to the collection is not evenly 
21   *     distributed (associative cache). The performance gain is about
22   *     50% for the fastest hash map collections (e.g. {@link FastMap}). 
23   *     For slower collections such as <code>java.util.TreeMap</code>, 
24   *     non-resizable {@link FastMap} (real-time) or database access,
25   *     performance can be of several orders of magnitude.</p>
26   * 
27   * <p> <b>Note:</b> The keys used to access elements of a {@link CachedMap} do
28   *     not need to be immutable as they are not stored in the cache
29   *     (only keys specified by the {@link #put} method are).
30   *     In other words, access can be performed using mutable keys as long
31   *     as these keys can be compared for equality with the real map's keys
32   *     (e.g. same <code>hashCode</code> values).</p>
33   * 
34   * <p> This implementation is not synchronized. Multiple threads accessing
35   *     or modifying the collection must be synchronized externally.</p>
36   *
37   * <p><i> This class is <b>public domain</b> (not copyrighted).</i></p>
38   * 
39   * @author  <a href="mailto:jean-marie@dautelle.com">Jean-Marie Dautelle</a>
40   * @version 5.3, October 30, 2003
41   */

85      /**
86       * Creates a cached map backed by a {@link FastMap} and having the 
87       * specified cache size.
88       *
89       * @param  cacheSize the cache size, the actual cache size is the
90       *         first power of 2 greater or equal to <code>cacheSize</code>.
91       *         This is also the initial capacity of the backing map.
92       */
93      public CachedMap(int cacheSize) {
94          this(cacheSize, new FastMap(cacheSize));
95      }

97      /**
98       * Creates a cached map backed by the specified map and having the specified
99       * cache size. In order to maitain cache veracity, it is critical
100      * that <b>all</b> update to the backing map is accomplished through the
101      * {@link CachedMap} instance; otherwise {@link #flush} has to be called.
102      *
103      * @param  cacheSize the cache size, the actual cache size is the
104      *         first power of 2 greater or equal to <code>cacheSize</code>.
105      * @param  backingMap the backing map to be "wrapped" in a cached map.
106      */
107     public CachedMap(int cacheSize, Map backingMap) {
108 
109         // Find a power of 2 >= minimalCache
110         int actualCacheSize = 1;
111         while (actualCacheSize < cacheSize) {
112             actualCacheSize <<= 1;
113         }
114 
115         // Sets up cache.
116         _keys = new Object[actualCacheSize];
117         _values = new Object[actualCacheSize];
118         _mask = actualCacheSize - 1;
119 
120         // Sets backing map references.
121         if (backingMap instanceof FastMap) {
122             _backingFastMap = (FastMap)backingMap;
123             _backingMap = _backingFastMap;
124             _keysMap = null;
125         } else {
126             _backingFastMap = null;
127             _backingMap = backingMap;
128             _keysMap = new FastMap(backingMap.size());
129             for ( Object key : backingMap.keySet() )
130             {
131                 _keysMap.put( key, key );
132             }
133         }
134     }

156     /**
157      * Flushes the key/value pairs being cached. This method should be called
158      * if the backing map is externally modified.
159      */
160     public void flush() {
161         for (int i=0; i < _keys.length; i++) {
162             _keys[i] = null;
163             _values[i] = null;
164         }
165 
166         if (_keysMap != null) {
167             // Re-populates keys from backing map.
168             for ( Object key : _backingMap.keySet() )
169             {
170                 _keysMap.put( key, key );
171             }
172         }
173     }

175     /**
176      * Returns the value to which this map maps the specified key.
177      * First, the cache is being checked, then if the cache does not contains
178      * the specified key, the backing map is accessed and the key/value
179      * pair is stored in the cache.
180      *
181      * @param  key the key whose associated value is to be returned.
182      * @return the value to which this map maps the specified key, or
183      *	       <code>null</code> if the map contains no mapping for this key.
184      * @throws ClassCastException if the key is of an inappropriate type for
185      *         the backing map (optional).
186      * @throws NullPointerException if the key is <code>null</code>.
187      */
188     public Object get(Object key) {
189         int index = key.hashCode() & _mask;
190         return key.equals(_keys[index]) ? 
191             _values[index] : getCacheMissed(key, index);
192     }
193     private Object getCacheMissed(Object key, int index) {
194         if (_backingFastMap != null) { 
195             Map.Entry entry = _backingFastMap.getEntry(key);
196             if (entry != null) {
197                 _keys[index] = entry.getKey();
198                 Object value = entry.getValue();
199                 _values[index] = value;
200                 return value;
201             } else {
202                 return null;
203             }
204         } else { // Generic backing map.
205             Object mapKey = _keysMap.get(key);
206             if (mapKey != null) {
207                 _keys[index] = mapKey;
208                 Object value = _backingMap.get(key);
209                 _values[index] = value;
210                 return value;
211             } else {
212                 return null;
213             }
214         }
215     }

217     /**
218      * Associates the specified value with the specified key in this map.
219      *
220      * @param  key the key with which the specified value is to be associated.
221      * @param  value the value to be associated with the specified key.
222      * @return the previous value associated with specified key, or
223      *         <code>null</code> if there was no mapping for the key.
224      * @throws UnsupportedOperationException if the <code>put</code> operation
225      *         is not supported by the backing map.
226      * @throws ClassCastException if the class of the specified key or value
227      * 	       prevents it from being stored in this map.
228      * @throws IllegalArgumentException if some aspect of this key or value
229      *	       prevents it from being stored in this map.
230      * @throws NullPointerException if the key is <code>null</code>.
231      */
232     public Object put(Object key, Object value) {
233         // Updates the cache.
234         int index = key.hashCode() & _mask;
235         if (key.equals(_keys[index]) ) {
236             _values[index] = value;
237         } else if (_keysMap != null) { // Possibly a new key.
238             _keysMap.put(key, key);
239         }
240 
241         // Updates the backing map.
242         return _backingMap.put(key, value);
243     }

245     /**
246      * Removes the mapping for this key from this map if it is present.
247      *
248      * @param  key key whose mapping is to be removed from the map.
249      * @return previous value associated with specified key,
250      *         or <code>null</code> if there was no mapping for key.
251      * @throws ClassCastException if the key is of an inappropriate type for
252      * 	       the backing map (optional).
253      * @throws NullPointerException if the key is <code>null</code>.
254      * @throws UnsupportedOperationException if the <code>remove</code> method
255      *         is not supported by the backing map.
256      */
257     public Object remove(Object key) {
258         // Removes from cache.
259         int index = key.hashCode() & _mask;
260         if (key.equals(_keys[index]) ) {
261             _keys[index] = null;
262         }
263         // Removes from key map.
264         if (_keysMap != null) {
265             _keysMap.remove(key);
266         }
267         // Removes from backing map.
268         return _backingMap.remove(key);
269     }

271     /**
272      * Indicates if this map contains a mapping for the specified key.
273      *
274      * @param   key the key whose presence in this map is to be tested.
275      * @return <code>true</code> if this map contains a mapping for the
276      *         specified key; <code>false</code> otherwise.
277      */
278     public boolean containsKey(Object key) {
279         // Checks the cache.
280         int index = key.hashCode() & _mask;
281         if (key.equals(_keys[index]) ) {
282             return true;
283         } else { // Checks the backing map.
284             return _backingMap.containsKey(key);
285         }
286     }
