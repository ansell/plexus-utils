33      /**
34       * Determines the relative path of a filename from a base directory.
35       * This method is useful in building relative links within pages of
36       * a web site.  It provides similar functionality to Anakia's
37       * <code>$relativePath</code> context variable.  The arguments to
38       * this method may contain either forward or backward slashes as
39       * file separators.  The relative path returned is formed using
40       * forward slashes as it is expected this path is to be used as a
41       * link in a web page (again mimicking Anakia's behavior).
42       * <p/>
43       * This method is thread-safe.
44       * <br/>
45       * <pre>
46       * PathTool.getRelativePath( null, null )                                   = ""
47       * PathTool.getRelativePath( null, "/usr/local/java/bin" )                  = ""
48       * PathTool.getRelativePath( "/usr/local/", null )                          = ""
49       * PathTool.getRelativePath( "/usr/local/", "/usr/local/java/bin" )         = ".."
50       * PathTool.getRelativePath( "/usr/local/", "/usr/local/java/bin/java.sh" ) = "../.."
51       * PathTool.getRelativePath( "/usr/local/java/bin/java.sh", "/usr/local/" ) = ""
52       * </pre>
53       *
54       * @param basedir  The base directory.
55       * @param filename The filename that is relative to the base
56       *                 directory.
57       * @return The relative path of the filename from the base
58       *         directory.  This value is not terminated with a forward slash.
59       *         A zero-length string is returned if: the filename is not relative to
60       *         the base directory, <code>basedir</code> is null or zero-length,
61       *         or <code>filename</code> is null or zero-length.
62       */
63      public static final String getRelativePath( String basedir, String filename )
64      {
65          basedir = uppercaseDrive( basedir );
66          filename = uppercaseDrive( filename );
67  
68          /*
69           * Verify the arguments and make sure the filename is relative
70           * to the base directory.
71           */
72          if ( basedir == null || basedir.length() == 0 || filename == null || filename.length() == 0
73              || !filename.startsWith( basedir ) )
74          {
75              return "";
76          }
77  
78          /*
79           * Normalize the arguments.  First, determine the file separator
80           * that is being used, then strip that off the end of both the
81           * base directory and filename.
82           */
83          String separator = determineSeparator( filename );
84          basedir = StringUtils.chompLast( basedir, separator );
85          filename = StringUtils.chompLast( filename, separator );
86  
87          /*
88           * Remove the base directory from the filename to end up with a
89           * relative filename (relative to the base directory).  This
90           * filename is then used to determine the relative path.
91           */
92          String relativeFilename = filename.substring( basedir.length() );
93  
94          return determineRelativePath( relativeFilename, separator );
95      }

97      /**
98       * Determines the relative path of a filename.  This method is
99       * useful in building relative links within pages of a web site.  It
100      * provides similar functionality to Anakia's
101      * <code>$relativePath</code> context variable.  The argument to
102      * this method may contain either forward or backward slashes as
103      * file separators.  The relative path returned is formed using
104      * forward slashes as it is expected this path is to be used as a
105      * link in a web page (again mimicking Anakia's behavior).
106      * <p/>
107      * This method is thread-safe.
108      *
109      * @param filename The filename to be parsed.
110      * @return The relative path of the filename. This value is not
111      *         terminated with a forward slash.  A zero-length string is
112      *         returned if: <code>filename</code> is null or zero-length.
113      * @see #getRelativeFilePath(String, String)
114      */
115     public static final String getRelativePath( String filename )
116     {
117         filename = uppercaseDrive( filename );
118 
119         if ( filename == null || filename.length() == 0 )
120         {
121             return "";
122         }
123 
124         /*
125          * Normalize the argument.  First, determine the file separator
126          * that is being used, then strip that off the end of the
127          * filename.  Then, if the filename doesn't begin with a
128          * separator, add one.
129          */
130 
131         String separator = determineSeparator( filename );
132         filename = StringUtils.chompLast( filename, separator );
133         if ( !filename.startsWith( separator ) )
134         {
135             filename = separator + filename;
136         }
137 
138         return determineRelativePath( filename, separator );
139     }

141     /**
142      * Determines the directory component of a filename.  This is useful
143      * within DVSL templates when used in conjunction with the DVSL's
144      * <code>$context.getAppValue("infilename")</code> to get the
145      * current directory that is currently being processed.
146      * <p/>
147      * This method is thread-safe.
148      * <br/>
149      * <pre>
150      * PathTool.getDirectoryComponent( null )                                   = ""
151      * PathTool.getDirectoryComponent( "/usr/local/java/bin" )                  = "/usr/local/java"
152      * PathTool.getDirectoryComponent( "/usr/local/java/bin/" )                 = "/usr/local/java/bin"
153      * PathTool.getDirectoryComponent( "/usr/local/java/bin/java.sh" )          = "/usr/local/java/bin"
154      * </pre>
155      *
156      * @param filename The filename to be parsed.
157      * @return The directory portion of the <code>filename</code>.  If
158      *         the filename does not contain a directory component, "." is
159      *         returned.
160      */
161     public static final String getDirectoryComponent( String filename )
162     {
163         if ( filename == null || filename.length() == 0 )
164         {
165             return "";
166         }
167 
168         String separator = determineSeparator( filename );
169         String directory = StringUtils.chomp( filename, separator );
170 
171         if ( filename.equals( directory ) )
172         {
173             return ".";
174         }
175 
176         return directory;
177     }

179     /**
180      * Calculates the appropriate link given the preferred link and the relativePath of the document.
181      * <br/>
182      * <pre>
183      * PathTool.calculateLink( "/index.html", "../.." )                                        = "../../index.html"
184      * PathTool.calculateLink( "http://plexus.codehaus.org/plexus-utils/index.html", "../.." ) = "http://plexus.codehaus.org/plexus-utils/index.html"
185      * PathTool.calculateLink( "/usr/local/java/bin/java.sh", "../.." )                        = "../../usr/local/java/bin/java.sh"
186      * PathTool.calculateLink( "../index.html", "/usr/local/java/bin" )                        = "/usr/local/java/bin/../index.html"
187      * PathTool.calculateLink( "../index.html", "http://plexus.codehaus.org/plexus-utils" )    = "http://plexus.codehaus.org/plexus-utils/../index.html"
188      * </pre>
189      *
190      * @param link
191      * @param relativePath
192      * @return String
193      */
194     public static final String calculateLink( String link, String relativePath )
195     {
196         if ( link == null )
197         {
198             link = "";
199         }
200         if ( relativePath == null )
201         {
202             relativePath = "";
203         }
204         //This must be some historical feature
205         if ( link.startsWith( "/site/" ) )
206         {
207             return link.substring( 5 );
208         }
209 
210         //Allows absolute links in nav-bars etc
211         if ( link.startsWith( "/absolute/" ) )
212         {
213             return link.substring( 10 );
214         }
215 
216         // This traps urls like http://
217         if ( link.contains( ":" ) )
218         {
219             return link;
220         }
221 
222         //If relativepath is current directory, just pass the link through
223         if ( StringUtils.equals( relativePath, "." ) )
224         {
225             if ( link.startsWith( "/" ) )
226             {
227                 return link.substring( 1 );
228             }
229 
230             return link;
231         }
232 
233         //If we don't do this, you can end up with ..//bob.html rather than ../bob.html
234         if ( relativePath.endsWith( "/" ) && link.startsWith( "/" ) )
235         {
236             return relativePath + "." + link.substring( 1 );
237         }
238 
239         if ( relativePath.endsWith( "/" ) || link.startsWith( "/" ) )
240         {
241             return relativePath + link;
242         }
243 
244         return relativePath + "/" + link;
245     }

247     /**
248      * This method can calculate the relative path between two pathes on a web site.
249      * <br/>
250      * <pre>
251      * PathTool.getRelativeWebPath( null, null )                                          = ""
252      * PathTool.getRelativeWebPath( null, "http://plexus.codehaus.org/" )                 = ""
253      * PathTool.getRelativeWebPath( "http://plexus.codehaus.org/", null )                 = ""
254      * PathTool.getRelativeWebPath( "http://plexus.codehaus.org/",
255      *                      "http://plexus.codehaus.org/plexus-utils/index.html" )        = "plexus-utils/index.html"
256      * PathTool.getRelativeWebPath( "http://plexus.codehaus.org/plexus-utils/index.html",
257      *                      "http://plexus.codehaus.org/"                                 = "../../"
258      * </pre>
259      *
260      * @param oldPath
261      * @param newPath
262      * @return a relative web path from <code>oldPath</code>.
263      */
264     public static final String getRelativeWebPath( final String oldPath, final String newPath )
265     {
266         if ( StringUtils.isEmpty( oldPath ) || StringUtils.isEmpty( newPath ) )
267         {
268             return "";
269         }
270 
271         String resultPath = buildRelativePath( newPath, oldPath, '/' );
272 
273         if ( newPath.endsWith( "/" ) && !resultPath.endsWith( "/" ) )
274         {
275             return resultPath + "/";
276         }
277 
278         return resultPath;
279     }

281     /**
282      * This method can calculate the relative path between two pathes on a file system.
283      * <br/>
284      * <pre>
285      * PathTool.getRelativeFilePath( null, null )                                   = ""
286      * PathTool.getRelativeFilePath( null, "/usr/local/java/bin" )                  = ""
287      * PathTool.getRelativeFilePath( "/usr/local", null )                           = ""
288      * PathTool.getRelativeFilePath( "/usr/local", "/usr/local/java/bin" )          = "java/bin"
289      * PathTool.getRelativeFilePath( "/usr/local", "/usr/local/java/bin/" )         = "java/bin"
290      * PathTool.getRelativeFilePath( "/usr/local/java/bin", "/usr/local/" )         = "../.."
291      * PathTool.getRelativeFilePath( "/usr/local/", "/usr/local/java/bin/java.sh" ) = "java/bin/java.sh"
292      * PathTool.getRelativeFilePath( "/usr/local/java/bin/java.sh", "/usr/local/" ) = "../../.."
293      * PathTool.getRelativeFilePath( "/usr/local/", "/bin" )                        = "../../bin"
294      * PathTool.getRelativeFilePath( "/bin", "/usr/local/" )                        = "../usr/local"
295      * </pre>
296      * Note: On Windows based system, the <code>/</code> character should be replaced by <code>\</code> character.
297      *
298      * @param oldPath
299      * @param newPath
300      * @return a relative file path from <code>oldPath</code>.
301      */
302     public static final String getRelativeFilePath( final String oldPath, final String newPath )
303     {
304         if ( StringUtils.isEmpty( oldPath ) || StringUtils.isEmpty( newPath ) )
305         {
306             return "";
307         }
308 
309         // normalise the path delimiters
310         String fromPath = new File( oldPath ).getPath();
311         String toPath = new File( newPath ).getPath();
312 
313         // strip any leading slashes if its a windows path
314         if ( toPath.matches( "^\\[a-zA-Z]:" ) )
315         {
316             toPath = toPath.substring( 1 );
317         }
318         if ( fromPath.matches( "^\\[a-zA-Z]:" ) )
319         {
320             fromPath = fromPath.substring( 1 );
321         }
322 
323         // lowercase windows drive letters.
324         if ( fromPath.startsWith( ":", 1 ) )
325         {
326             fromPath = Character.toLowerCase( fromPath.charAt( 0 ) ) + fromPath.substring( 1 );
327         }
328         if ( toPath.startsWith( ":", 1 ) )
329         {
330             toPath = Character.toLowerCase( toPath.charAt( 0 ) ) + toPath.substring( 1 );
331         }
332 
333         // check for the presence of windows drives. No relative way of
334         // traversing from one to the other.
335         if ( ( toPath.startsWith( ":", 1 ) && fromPath.startsWith( ":", 1 ) ) && ( !toPath.substring( 0, 1 ).equals(
336             fromPath.substring( 0, 1 ) ) ) )
337         {
338             // they both have drive path element but they dont match, no
339             // relative path
340             return null;
341         }
342 
343         if ( ( toPath.startsWith( ":", 1 ) && !fromPath.startsWith( ":", 1 ) ) || ( !toPath.startsWith( ":", 1 )
344             && fromPath.startsWith( ":", 1 ) ) )
345         {
346             // one has a drive path element and the other doesnt, no relative
347             // path.
348             return null;
349         }
350 
351         String resultPath = buildRelativePath( toPath, fromPath, File.separatorChar );
352 
353         if ( newPath.endsWith( File.separator ) && !resultPath.endsWith( File.separator ) )
354         {
355             return resultPath + File.separator;
356         }
357 
358         return resultPath;
359     }

365     /**
366      * Determines the relative path of a filename.  For each separator
367      * within the filename (except the leading if present), append the
368      * "../" string to the return value.
369      *
370      * @param filename  The filename to parse.
371      * @param separator The separator used within the filename.
372      * @return The relative path of the filename.  This value is not
373      *         terminated with a forward slash.  A zero-length string is
374      *         returned if: the filename is zero-length.
375      */
376     private static final String determineRelativePath( String filename, String separator )
377     {
378         if ( filename.length() == 0 )
379         {
380             return "";
381         }
382 
383         /*
384         * Count the slashes in the relative filename, but exclude the
385         * leading slash.  If the path has no slashes, then the filename
386         * is relative to the current directory.
387         */
388         int slashCount = StringUtils.countMatches( filename, separator ) - 1;
389         if ( slashCount <= 0 )
390         {
391             return ".";
392         }
393 
394         /*
395          * The relative filename contains one or more slashes indicating
396          * that the file is within one or more directories.  Thus, each
397          * slash represents a "../" in the relative path.
398          */
399         StringBuilder sb = new StringBuilder();
400         for ( int i = 0; i < slashCount; i++ )
401         {
402             sb.append( "../" );
403         }
404 
405         /*
406          * Finally, return the relative path but strip the trailing
407          * slash to mimic Anakia's behavior.
408          */
409         return StringUtils.chop( sb.toString() );
410     }

412     /**
413      * Helper method to determine the file separator (forward or
414      * backward slash) used in a filename.  The slash that occurs more
415      * often is returned as the separator.
416      *
417      * @param filename The filename parsed to determine the file
418      *                 separator.
419      * @return The file separator used within <code>filename</code>.
420      *         This value is either a forward or backward slash.
421      */
422     private static final String determineSeparator( String filename )
423     {
424         int forwardCount = StringUtils.countMatches( filename, "/" );
425         int backwardCount = StringUtils.countMatches( filename, "\\" );
426 
427         return forwardCount >= backwardCount ? "/" : "\\";
428     }

430     /**
431      * Cygwin prefers lowercase drive letters, but other parts of maven use uppercase
432      *
433      * @param path
434      * @return String
435      */
436     static final String uppercaseDrive( String path )
437     {
438         if ( path == null )
439         {
440             return null;
441         }
442         if ( path.length() >= 2 && path.charAt( 1 ) == ':' )
443         {
444             path = Character.toUpperCase( path.charAt( 0 ) ) + path.substring( 1 );
445         }
446         return path;
447     }

449     private static final String buildRelativePath( String toPath, String fromPath, final char separatorChar )
450     {
451         // use tokeniser to traverse paths and for lazy checking
452         StringTokenizer toTokeniser = new StringTokenizer( toPath, String.valueOf( separatorChar ) );
453         StringTokenizer fromTokeniser = new StringTokenizer( fromPath, String.valueOf( separatorChar ) );
454 
455         int count = 0;
456 
457         // walk along the to path looking for divergence from the from path
458         while ( toTokeniser.hasMoreTokens() && fromTokeniser.hasMoreTokens() )
459         {
460             if ( separatorChar == '\\' )
461             {
462                 if ( !fromTokeniser.nextToken().equalsIgnoreCase( toTokeniser.nextToken() ) )
463                 {
464                     break;
465                 }
466             }
467             else
468             {
469                 if ( !fromTokeniser.nextToken().equals( toTokeniser.nextToken() ) )
470                 {
471                     break;
472                 }
473             }
474 
475             count++;
476         }
477 
478         // reinitialise the tokenisers to count positions to retrieve the
479         // gobbled token
480 
481         toTokeniser = new StringTokenizer( toPath, String.valueOf( separatorChar ) );
482         fromTokeniser = new StringTokenizer( fromPath, String.valueOf( separatorChar ) );
483 
484         while ( count-- > 0 )
485         {
486             fromTokeniser.nextToken();
487             toTokeniser.nextToken();
488         }
489 
490         String relativePath = "";
491 
492         // add back refs for the rest of from location.
493         while ( fromTokeniser.hasMoreTokens() )
494         {
495             fromTokeniser.nextToken();
496 
497             relativePath += "..";
498 
499             if ( fromTokeniser.hasMoreTokens() )
500             {
501                 relativePath += separatorChar;
502             }
503         }
504 
505         if ( relativePath.length() != 0 && toTokeniser.hasMoreTokens() )
506         {
507             relativePath += separatorChar;
508         }
509 
510         // add fwd fills for whatevers left of newPath.
511         while ( toTokeniser.hasMoreTokens() )
512         {
513             relativePath += toTokeniser.nextToken();
514 
515             if ( toTokeniser.hasMoreTokens() )
516             {
517                 relativePath += separatorChar;
518             }
519         }
520         return relativePath;
521     }
