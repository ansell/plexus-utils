18  public class EntityReplacementMap

20      final String entityName[];

27      public EntityReplacementMap( String[][] replacements ){
28          int length = replacements.length;
29          entityName = new String[length];
30          entityNameBuf = new char[length][];
31          entityReplacement = new String[length];
32          entityReplacementBuf = new char[length][];
33          entityNameHash = new int[length];
34  
35          for ( String[] replacement : replacements )
36          {
37              defineEntityReplacementText( replacement[0],replacement[1] );
38          }
39      }

41      private void defineEntityReplacementText(String entityName,
42                                              String replacementText)
43      {
44          if ( !replacementText.startsWith( "&#" ) && this.entityName != null && replacementText.length() > 1 )
45          {
46              String tmp = replacementText.substring( 1, replacementText.length() - 1 );
47              for ( int i = 0; i < this.entityName.length; i++ )
48              {
49                  if ( this.entityName[i] != null && this.entityName[i].equals( tmp ) )
50                  {
51                      replacementText = this.entityReplacement[i];
52                  }
53              }
54          }
55  
56          // this is to make sure that if interning works we will take advantage of it ...
57          char[] entityNameCharData = entityName.toCharArray();
58          //noinspection ConstantConditions
59          this.entityName[entityEnd] = newString( entityNameCharData, 0, entityName.length());
60          entityNameBuf[entityEnd] = entityNameCharData;
61  
62          entityReplacement[entityEnd] = replacementText;
63          entityReplacementBuf[entityEnd] = replacementText.toCharArray();
64          entityNameHash[ entityEnd ] = fastHash( entityNameBuf[entityEnd], 0, entityNameBuf[entityEnd].length);
65          ++entityEnd;
66          //TODO disallow < or & in entity replacement text (or ]]>???)
67          //TODO keepEntityNormalizedForAttributeValue cached as well ...
68      }

70      private String newString(char[] cbuf, int off, int len) {
71          return new String(cbuf, off, len);
72      }

74      /**
75       * simplistic implementation of hash function that has <b>constant</b>
76       * time to compute - so it also means diminishing hash quality for long strings
77       * but for XML parsing it should be good enough ...
78       */
79      private static int fastHash( char ch[], int off, int len ) {
80          if(len == 0) return 0;
81          //assert len >0
82          int hash = ch[off]; // hash at beginning
83          //try {
84          hash = (hash << 7) + ch[ off +  len - 1 ]; // hash at the end
85          //} catch(ArrayIndexOutOfBoundsException aie) {
86          //    aie.printStackTrace(); //should never happen ...
87          //    throw new RuntimeException("this is violation of pre-condition");
88          //}
89          if(len > 16) hash = (hash << 7) + ch[ off + (len / 4)];  // 1/4 from beginning
90          if(len > 8)  hash = (hash << 7) + ch[ off + (len / 2)];  // 1/2 of string size ...
91          // notice that hash is at most done 3 times <<7 so shifted by 21 bits 8 bit value
92          // so max result == 29 bits so it is quite just below 31 bits for long (2^32) ...
93          //assert hash >= 0;
94          return  hash;
95      }
