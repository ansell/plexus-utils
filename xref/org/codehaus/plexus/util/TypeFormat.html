8   /**
9    * <p> This class provides utility methods to parse <code>CharSequence</code>
10   *     into primitive types and to format primitive types into
11   *     <code>StringBuffer</code>.</p>
12   *
13   * <p> Methods from this utility class <b>do not create temporary objects</b>
14   *     and are typically faster than standard library methods (e.g {@link
15   *     #parseDouble} is up to 15x faster than <code>Double.parseDouble</code>).
16   *     </p>
17   *
18   * <p> For class instances, formatting is typically performed using specialized
19   *     <code>java.text.Format</code> (<code>Locale</code> sensitive)
20   *     and/or using conventional methods (class sensitive). For example:<pre>
21   *     public class Foo {
22   *         public static Foo valueOf(CharSequence chars) {...} // Parses.
23   *         public StringBuffer appendTo(StringBuffer sb) {...} // Formats.
24   *         public String toString() {
25   *             return appendTo(new StringBuffer()).toString();
26   *         }
27   *     }</pre></p>
28   *
29   * <p><i> This class is <b>public domain</b> (not copyrighted).</i></p>
30   *
31   * @author  <a href="mailto:jean-marie@dautelle.com">Jean-Marie Dautelle</a>
32   * @version 4.6, June 22, 2003

50      /**
51       * Searches for a particular sequence within a character sequence
52       * (general purpose parsing function).
53       *
54       * @param  pattern the character sequence to search for.
55       * @param  chars the character sequence being searched.
56       * @param  fromIndex the index in <code>chars</code> to start the search
57       *         from.
58       * @return the index in the range
59       *         <code>[fromIndex, chars.length()-pattern.length()]</code>
60       *         or <code>-1</code> if the character sequence is not found.
61       */
62      public static int indexOf(CharSequence pattern, CharSequence chars,
63                               int fromIndex) {
64          int patternLength = pattern.length();
65          fromIndex = Math.max(0, fromIndex);
66          if (patternLength != 0) { // At least one character to search for.
67              char firstChar = pattern.charAt(0);
68              int last = chars.length() - patternLength;
69              for (int i=fromIndex; i <= last; i++) {
70                  if (chars.charAt(i) == firstChar) {
71                      boolean match = true;
72                      for (int j=1; j < patternLength; j++) {
73                          if (chars.charAt(i+j) != pattern.charAt(j)) {
74                              match = false;
75                              break;
76                          }
77                      }
78                      if (match) {
79                          return i;
80                      }
81                  }
82              }
83              return -1;
84          } else {
85              return Math.min(0, fromIndex);
86          }
87      }

117     /**
118      * Parses the specified <code>CharSequence</code> as a signed
119      * <code>short</code> in the specified radix. The characters in the string
120      * must all be digits of the specified radix, except the first character
121      * which may be a plus sign <code>'+'</code> or a minus sign
122      * <code>'-'</code>.
123      *
124      * @param  chars the character sequence to parse.
125      * @param  radix the radix to be used while parsing.
126      * @return the corresponding <code>short</code>.
127      * @throws NumberFormatException if the specified character sequence
128      *         does not contain a parsable <code>short</code>.
129      */
130     public static short parseShort(CharSequence chars, int radix) {
131         try {
132             boolean isNegative = (chars.charAt(0) == '-') ? true : false;
133             int result = 0;
134             int limit = (isNegative) ? Short.MIN_VALUE : -Short.MAX_VALUE;
135             int multmin = limit / radix;
136             int length = chars.length();
137             int i = (isNegative || (chars.charAt(0) == '+')) ? 1 : 0;
138             while (true) {
139                 int digit = Character.digit(chars.charAt(i), radix);
140                 int tmp = result * radix;
141                 if ((digit < 0) || (result < multmin) ||
142                                    (tmp < limit + digit)) { // Overflow.
143                     throw new NumberFormatException(
144                         "For input characters: \"" + chars.toString() + "\"");
145                 }
146                 // Accumulates negatively.
147                 result = tmp - digit;
148                 if (++i >= length) {
149                     break;
150                 }
151             }
152             return (short) (isNegative ? result : -result);
153         } catch (IndexOutOfBoundsException e) {
154             throw new NumberFormatException(
155                   "For input characters: \"" + chars.toString() + "\"");
156         }
157     }

290     /**
291      * Parses this <code>CharSequence</code> as a <code>double</code>.
292      *
293      * @param  chars the character sequence to parse.
294      * @return the double number represented by this character sequence.
295      * @throws NumberFormatException if the character sequence does not contain
296      *         a parsable <code>double</code>.
297      */
298     public static double parseDouble(CharSequence chars)
299             throws NumberFormatException {
300         try {
301             int length = chars.length();
302             double result = 0.0;
303             int exp = 0;
304 
305             boolean isNegative = (chars.charAt(0) == '-') ? true : false;
306             int i = (isNegative || (chars.charAt(0) == '+')) ? 1 : 0;
307 
308             // Checks special cases NaN or Infinity.
309             if ((chars.charAt(i) == 'N') || (chars.charAt(i) == 'I')) {
310                 if (chars.toString().equals("NaN")) {
311                     return Double.NaN;
312                 } else if (chars.subSequence(i, length).toString().
313                         equals("Infinity")) {
314                     return isNegative ? Double.NEGATIVE_INFINITY :
315                                         Double.POSITIVE_INFINITY;
316                 }
317             }
318 
319             // Reads decimal number.
320             boolean fraction = false;
321             while (true) {
322                 char c = chars.charAt(i);
323                 if ( (c == '.') && (!fraction)) {
324                     fraction = true;
325                 } else if ((c == 'e') || (c == 'E')) {
326                     break;
327                 } else if ((c >= '0') && (c <= '9')) {
328                     result = result * 10 + (c - '0');
329                     if (fraction) {
330                         exp--;
331                     }
332                 } else {
333                     throw new NumberFormatException(
334                         "For input characters: \"" + chars.toString() + "\"");
335                 }
336                 if (++i >= length) {
337                     break;
338                 }
339             }
340             result = isNegative ? - result : result;
341 
342             // Reads exponent (if any).
343             if (i < length) {
344                 i++;
345                 boolean negE = (chars.charAt(i) == '-') ? true : false;
346                 i = (negE || (chars.charAt(i) == '+')) ? i+1 : i;
347                 int valE = 0;
348                 while (true) {
349                     char c = chars.charAt(i);
350                     if ((c >= '0') && (c <= '9')) {
351                         valE = valE * 10 + (c - '0');
352                         if (valE > 10000000) { // Hard-limit to avoid overflow.
353                             valE = 10000000;
354                         }
355                     } else {
356                         throw new NumberFormatException(
357                             "For input characters: \"" + chars.toString() +
358                              "\"");
359                     }
360                     if (++i >= length) {
361                         break;
362                     }
363                 }
364                 exp += negE ? -valE : valE;
365             }
366 
367             // Returns product decimal number with exponent.
368             return multE(result, exp);
369 
370         } catch (IndexOutOfBoundsException e) {
371             throw new NumberFormatException(
372                   "For input characters: \"" + chars.toString() + "\"");
373         }
374     }

420     /**
421      * Formats the specified <code>int</code> and appends the resulting
422      * text (decimal representation) to the <code>StringBuffer</code> argument.
423      *
424      * <p> Note: This method is preferred to <code>StringBuffer.append(int)
425      *           </code> as it does not create temporary <code>String</code>
426      *           objects (several times faster for small numbers).</p>
427      *
428      * @param  i the <code>int</code> number.
429      * @param  sb the <code>StrinBuffer</code> to append.
430      * @return the specified <code>StringBuffer</code> object.
431      * @see    #parseInt
432      */
433     public static StringBuffer format(int i, StringBuffer sb) {
434         if (i <= 0) {
435             if (i == Integer.MIN_VALUE) { // Negation would overflow.
436                 return sb.append("-2147483648"); // 11 char max.
437             } else if (i == 0) {
438                 return sb.append('0');
439             }
440             i = -i;
441             sb.append('-');
442         }
443         int j = 1;
444         for (; (j < 10) && (i >= INT_POW_10[j]); j++) {}
445         // POW_10[j] > i >= POW_10[j-1]
446         for (j--; j >= 0; j--) {
447             int pow10 = INT_POW_10[j];
448             int digit = i / pow10;
449             i -= digit * pow10;
450             sb.append(DIGITS[digit]);
451         }
452         return sb;
453     }
454     private static final int[] INT_POW_10 = new int[10];
455     static {
456         int pow = 1;
457         for (int i=0; i < 10; i++) {
458             INT_POW_10[i] = pow;
459             pow *= 10;
460         }
461     }

489         if (i <= -radix) {
490             format2(i / radix, radix, sb);
491             sb.append(DIGITS[-(i % radix)]);
492         } else {
493             sb.append(DIGITS[-i]);
494         }

587     /**
588      * Formats the specified <code>float</code> and appends the resulting text
589      * to the <code>StringBuffer</code> argument; the number of significative
590      * digits is deduced from the specifed precision. All digits at least as
591      * significant as the specified precision are represented. For example:
592      * <ul>
593      * <li><code>format(5.6f, 0.01f, sb)</code> appends <code>"5.60"</code></li>
594      * <li><code>format(5.6f, 0.1f, sb)</code> appends <code>"5.6"</code></li>
595      * <li><code>format(5.6f, 1f, sb)</code> appends <code>"6"</code></li>
596      * </ul>
597      * If the precision is <code>0.0f</code>, the precision is assumed to be
598      * the intrinsic <code>float</code> precision (64 bits IEEE 754 format);
599      * no formatting is performed, all significant digits are displayed and
600      * trailing zeros are removed.
601      *
602      * @param  f the <code>float</code> number.
603      * @param  precision the maximum weight of the last digit represented.
604      * @param  sb the <code>StrinBuffer</code> to append.
605      * @return the specified <code>StringBuffer</code> object.
606      * @throws IllegalArgumentException if the specified precision is negative
607      *         or would result in too many digits (19+).
608      */
609     public static StringBuffer format(float f, float precision,
610                                       StringBuffer sb) {
611         // Adjusts precision.
612         boolean precisionOnLastDigit;
613         if (precision > 0.0f) {
614             precisionOnLastDigit = true;
615         } else if (precision == 0.0f) {

674      * to the <code>StringBuffer</code> argument; the number of significative
675      * digits is deduced from the specifed precision. All digits at least as
676      * significant as the specified precision are represented. For example:
677      * <ul>
678      * <li><code>format(5.6, 0.01, sb)</code> appends <code>"5.60"</code></li>
679      * <li><code>format(5.6, 0.1, sb)</code> appends <code>"5.6"</code></li>
680      * <li><code>format(5.6, 1, sb)</code> appends <code>"6"</code></li>
681      * </ul>
682      * If the precision is <code>0.0</code>, the precision is assumed to be
683      * the intrinsic <code>double</code> precision (64 bits IEEE 754 format);
684      * no formatting is performed, all significant digits are displayed and
685      * trailing zeros are removed.
686      *
687      * @param  d the <code>double</code> number.
688      * @param  precision the maximum weight of the last digit represented.
689      * @param  sb the <code>StrinBuffer</code> to append.
690      * @return the specified <code>StringBuffer</code> object.
691      * @throws IllegalArgumentException if the specified precision is negative
692      *         or would result in too many digits (19+).
693      */
694     public static StringBuffer format(double d, double precision,
695                                       StringBuffer sb) {
696         // Adjusts precision.
697         boolean precisionOnLastDigit = false;
698         if (precision > 0.0) {
699             precisionOnLastDigit = true;
700         } else if (precision == 0.0) {
701             if (d != 0.0) {
702                 precision = Math.max(Math.abs(d * DOUBLE_RELATIVE_ERROR),
703                                      Double.MIN_VALUE);
704             } else {
705                 return sb.append("0.0"); // Exact zero.
706             }
707         } else if (precision < 0.0) { // Not NaN
708             throw new IllegalArgumentException(
709                "precision: Negative values not allowed");
710         }
711         return format(d, precision, precisionOnLastDigit, sb);
712     }

714     /**
715      * Formats the specified <code>double</code> and appends the resulting text
716      * to the <code>StringBuffer</code> argument; the number of significative
717      * digits is deduced from the specifed precision.
718      *
719      * @param  d the <code>double</code> number.
720      * @param  precision the maximum weight of the last digit represented.
721      * @param  precisionOnLastDigit indicates if the number of digits is
722      *         deduced from the specified precision.
723      * @param  sb the <code>StrinBuffer</code> to append.
724      * @return the specified <code>StringBuffer</code> object.
725      */
726     private static StringBuffer format(double d, double precision,
727                                        boolean precisionOnLastDigit,
728                                       StringBuffer sb) {
729         // Special cases.
730         if (Double.isNaN(d)) {
731             return sb.append("NaN");
732         } else if (Double.isInfinite(d)) {
733             return (d >= 0) ? sb.append("Infinity") : sb.append("-Infinity");
734         }
735         if (d < 0) {
736             d = -d;
737             sb.append('-');
738         }
739 
740         // Formats decimal part.
741         int rank = (int) Math.floor(Math.log(precision) / LOG_10);
742         double digitValue = multE(d, -rank);
743         if (digitValue >= Long.MAX_VALUE) {
744             throw new IllegalArgumentException(
745                "Specified precision would result in too many digits");
746         }
747         int digitStart = sb.length();
748         format(Math.round(digitValue), sb);
749         int digitLength = sb.length() - digitStart;
750         int dotPos = digitLength + rank;
751         boolean useScientificNotation = false;
752 
753         // Inserts dot ('.')
754         if ((dotPos <= -LEADING_ZEROS.length) || (dotPos > digitLength)) {
755             // Scientific notation has to be used ("x.xxxEyy").
756             sb.insert(digitStart + 1, '.');
757             useScientificNotation = true;
758         } else if (dotPos > 0) {
759             //  Dot within the string ("xxxx.xxxxx").
760             sb.insert(digitStart + dotPos, '.');
761         } else {
762             // Leading zeros ("0.xxxxx").
763             sb.insert(digitStart, LEADING_ZEROS[-dotPos]);
764         }
765 
766         // Removes trailing zeros.
767         if (!precisionOnLastDigit) {
768             int newLength = sb.length();
769             do {
770                 newLength--;
771             } while (sb.charAt(newLength) == '0');
772             sb.setLength(newLength+1);
773         }
774 
775         // Avoids trailing '.'
776         if (sb.charAt(sb.length()-1) == '.') {
777             if (precisionOnLastDigit) {
778                 sb.setLength(sb.length()-1); // Prefers "xxx" to "xxx."
779             } else {
780                 sb.append('0'); // Prefer "xxx.0" to "xxx."
781             }
782         }
783 
784         // Writes exponent.
785         if (useScientificNotation) {
786             sb.append('E');
787             format(dotPos - 1, sb);
788         }
789 
790         return sb;
791     }

793     private static final float FLOAT_RELATIVE_ERROR = (float) Math.pow(2, -24);
794     private static final double DOUBLE_RELATIVE_ERROR = Math.pow(2, -53);
795     private static String[] LEADING_ZEROS = {"0.", "0.0", "0.00"};

799      * Returns the product of the specified value with <code>10</code> raised
800      * at the specified power exponent.
801      *
802      * @param  value the value.
803      * @param  E the exponent.
804      * @return <code>value * 10^E</code>
805      */
806     private static final double multE(double value, int E) {
807         if (E >= 0) {
808            if (E <= 308) {
809                 // Max: 1.7976931348623157E+308
810                return value * DOUBLE_POW_10[E];
811             } else {
812                 value *= 1E21; // Exact multiplicand.
813                 E = Math.min(308, E-21);
814                 return value * DOUBLE_POW_10[E];
815             }
816         } else {
817             if (E >= -308) {
818                 return value / DOUBLE_POW_10[-E];
819             } else {
820                 // Min: 4.9E-324
821                 value /= 1E21; // Exact divisor.
822                 E = Math.max(-308, E+21);
823                 return value / DOUBLE_POW_10[-E];
824             }
825         }
